## coté serveur

sudo git clone https://github.com/Nyr/openvpn-install.git
cd openvpn-install/
sudo bash openvpn-install.sh

##########

## coté client

sudo apt-get install openvpn -y
cd /etc/openvpn
sudo wget .conf
sudo mv client.ovpn client.conf
sudo nano /etc/default/openvpn
autostart="all"
sudo systemctl daemon-reload
sudo systemctl start openvpn

## script de reconnection

cd /etc/shinken/
sudo nano reconnect.sh

#!/bin/sh
ps auxw | grep openvpn | grep -v grep > /dev/null
if [ $? != 0 ]
then
        /etc/init.d/openvpn start > /dev/null
fi

sudo chmod +x reconnect.sh

sudo crontab -e

*/1 * * * * /etc/shinken/reconnect.sh
